เกมงู (Snake Game)

ใช้ปุ่มลูกศรเพื่อควบคุมงู กินอาหารให้ได้มากที่สุด!

คะแนน: 0

กดปุ่มลูกศรเพื่อเริ่มเกม

import turtle import random import time # ตั้งค่า turtle screen = turtle.Screen() screen.title("Snake Game") screen.bgcolor("black") screen.setup(width=600, height=600) screen.tracer(0) # ปิดการอัปเดตอัตโนมัติ # งู snake = [] snake_speed = 0.1 snake_direction = "stop" # อาหาร food = turtle.Turtle() food.speed(0) food.shape("circle") food.color("red") food.penup() food.goto(0, 100) # คะแนน score = 0 high_score = 0 # แสดงคะแนน pen = turtle.Turtle() pen.speed(0) pen.shape("square") pen.color("white") pen.penup() pen.hideturtle() pen.goto(0, 260) pen.write("Score: 0 High Score: 0", align="center", font=("Courier", 24, "normal")) # ฟังก์ชันควบคุมทิศทาง def go_up(): global snake_direction if snake_direction != "down": snake_direction = "up" def go_down(): global snake_direction if snake_direction != "up": snake_direction = "down" def go_left(): global snake_direction if snake_direction != "right": snake_direction = "left" def go_right(): global snake_direction if snake_direction != "left": snake_direction = "right" # ฟังก์ชันเคลื่อนที่ def move(): if snake_direction == "up": y = snake[0].ycor() snake[0].sety(y + 20) if snake_direction == "down": y = snake[0].ycor() snake[0].sety(y - 20) if snake_direction == "left": x = snake[0].xcor() snake[0].setx(x - 20) if snake_direction == "right": x = snake[0].xcor() snake[0].setx(x + 20) # ฟังก์ชันหลัก def game_loop(): global score, high_score, snake_direction # เคลื่อนที่งู move() # ตรวจสอบการชนขอบ head = snake[0] if head.xcor() > 290 or head.xcor() < -290 or head.ycor() > 290 or head.ycor() < -290: game_over() # ตรวจสอบการชนตัวเอง for segment in snake[1:]: if head.distance(segment) < 20: game_over() # ตรวจสอบการกินอาหาร if head.distance(food) < 20: # ย้ายอาหาร x = random.randint(-290, 290) y = random.randint(-290, 290) food.goto(x, y) # เพิ่มส่วนงู new_segment = turtle.Turtle() new_segment.speed(0) new_segment.shape("square") new_segment.color("green") new_segment.penup() snake.append(new_segment) # เพิ่มคะแนน score += 10 if score > high_score: high_score = score pen.clear() pen.write("Score: {} High Score: {}".format(score, high_score), align="center", font=("Courier", 24, "normal")) # อัปเดต DOM document.getElementById("score").innerText = score # เคลื่อนที่ส่วนอื่นๆ ของงู for i in range(len(snake)-1, 0, -1): x = snake[i-1].xcor() y = snake[i-1].ycor() snake[i].goto(x, y) # อัปเดตหน้าจอ screen.update() # เรียกตัวเองอีกครั้ง screen.ontimer(game_loop, int(snake_speed * 1000)) def game_over(): global snake_direction snake_direction = "stop" pen.goto(0, 0) pen.write("Game Over", align="center", font=("Courier", 36, "normal")) time.sleep(2) pen.clear() pen.goto(0, 260) pen.write("Score: {} High Score: {}".format(score, high_score), align="center", font=("Courier", 24, "normal")) # เริ่มเกม def start_game(): global snake, score score = 0 pen.clear() pen.write("Score: 0 High Score: {}".format(high_score), align="center", font=("Courier", 24, "normal")) # ล้างงูเก่า for segment in snake: segment.goto(1000, 1000) snake.clear() # สร้างงูใหม่ head = turtle.Turtle() head.speed(0) head.shape("square") head.color("green") head.penup() head.goto(0, 0) snake.append(head) # ตั้งค่าปุ่มควบคุม screen.listen() screen.onkey(go_up, "Up") screen.onkey(go_down, "Down") screen.onkey(go_left, "Left") screen.onkey(go_right, "Right") # เริ่มเกม game_loop() # เรียกเริ่มเกมเมื่อโหลด start_game()
